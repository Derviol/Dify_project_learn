### 3.3 Dify 服务封装 (services/dify_service.py)

```python
import httpx
from app.core.config import settings

class DifyService:
    """封装与 Dify 平台的所有交互"""

    def __init__(self):
        self.client = httpx.AsyncClient(timeout=60.0)
        self.app_keys = {
            "planner": settings.DIFY_APP_KEY_PLANNER,
            "guide": settings.DIFY_APP_KEY_GUIDE,
            "ticket": settings.DIFY_APP_KEY_TICKET,
        }

    async def chat(
        self,
        app_type: str,        # planner / guide / ticket
        query: str,           # 用户消息
        user_id: str,         # 用户标识
        conversation_id: str = None,  # 会话ID（续接对话）
        inputs: dict = None,  # 输入变量
        stream: bool = True,  # 是否流式
    ) -> AsyncGenerator:
        """调用 Dify Chat API（流式）"""

        app_key = self.app_keys[app_type]
        url = f"{settings.DIFY_API_BASE}/chat-messages"

        payload = {
            "inputs": inputs or {},
            "query": query,
            "response_mode": "streaming" if stream else "blocking",
            "user": user_id,
        }
        if conversation_id:
            payload["conversation_id"] = conversation_id

        headers = {
            "Authorization": f"Bearer {app_key}",
            "Content-Type": "application/json",
        }

        if stream:
            async with self.client.stream("POST", url, json=payload, headers=headers) as resp:
                async for line in resp.aiter_lines():
                    if line.startswith("data: "):
                        yield line[6:]  # SSE 格式
        else:
            resp = await self.client.post(url, json=payload, headers=headers)
            yield resp.json()

    async def run_workflow(
        self,
        workflow_id: str,
        inputs: dict,
        user_id: str,
        response_mode: str = "blocking",
    ) -> dict:
        """调用 Dify Workflow 执行 API"""

        url = f"{settings.DIFY_API_BASE}/workflows/run"
        payload = {
            "inputs": inputs,
            "response_mode": response_mode,
            "user": user_id,
        }
        headers = {
            "Authorization": f"Bearer {workflow_id}",
            "Content-Type": "application/json",
        }
        resp = await self.client.post(url, json=payload, headers=headers)
        return resp.json()

dify_service = DifyService()
```

### 3.4 排队服务 (services/queue_service.py)

```python
import redis.asyncio as redis
from datetime import datetime

class QueueService:
    """基于 Redis 的实时排队管理"""

    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client

    async def get_queue_info(self, project_id: str) -> dict:
        """获取项目实时排队信息"""
        key = f"queue:{project_id}"
        data = await self.redis.hgetall(key)
        return {
            "project_id": project_id,
            "queue_length": int(data.get("length", 0)),
            "estimated_wait": int(data.get("wait_minutes", 0)),
            "status": data.get("status", "unknown"),
            "crowd_level": data.get("crowd_level", "适中"),
            "updated_at": data.get("updated_at"),
        }

    async def take_number(self, project_id: str, user_id: str) -> dict:
        """线上取号"""
        key = f"queue:{project_id}:numbers"
        number = await self.redis.incr(f"queue:{project_id}:counter")
        ticket_data = {
            "number": number,
            "user_id": user_id,
            "taken_at": datetime.now().isoformat(),
            "status": "waiting",
        }
        await self.redis.hset(f"queue:{project_id}:ticket:{number}", mapping=ticket_data)
        await self.redis.expire(f"queue:{project_id}:ticket:{number}", 7200)  # 2小时过期
        return ticket_data

    async def get_my_number(self, project_id: str, user_id: str) -> dict | None:
        """查询我的排队状态"""
        # 查找用户的排队号
        pattern = f"queue:{project_id}:ticket:*"
        async for key in self.redis.scan_iter(match=pattern):
            data = await self.redis.hgetall(key)
            if data.get("user_id") == user_id and data.get("status") == "waiting":
                current = await self.redis.get(f"queue:{project_id}:current")
                ahead = int(data["number"]) - int(current or 0)
                return {
                    "number": int(data["number"]),
                    "ahead_count": max(0, ahead),
                    "estimated_wait": max(0, ahead * 3),  # 约3分钟/号
                }
        return None

queue_service = None  # 在 main.py 初始化时注入 redis client
```

### 3.5 中间件 (core/middleware.py)

```python
from starlette.middleware.base import BaseHTTPMiddleware
from fastapi import Request
from fastapi.responses import JSONResponse
import time
import uuid

class RequestIDMiddleware(BaseHTTPMiddleware):
    """为每个请求分配唯一 ID"""
    async def dispatch(self, request: Request, call_next):
        request_id = str(uuid.uuid4())[:8]
        request.state.request_id = request_id
        response = await call_next(request)
        response.headers["X-Request-ID"] = request_id
        return response

class TimingMiddleware(BaseHTTPMiddleware):
    """记录请求耗时"""
    async def dispatch(self, request: Request, call_next):
        start = time.perf_counter()
        response = await call_next(request)
        elapsed = time.perf_counter() - start
        response.headers["X-Process-Time"] = f"{elapsed:.4f}"
        return response

class RateLimitMiddleware(BaseHTTPMiddleware):
    """基于 Redis 的 API 限流"""
    def __init__(self, app, redis_client, max_requests=60, window=60):
        super().__init__(app)
        self.redis = redis_client
        self.max_requests = max_requests
        self.window = window

    async def dispatch(self, request: Request, call_next):
        client_ip = request.client.host
        key = f"rate_limit:{client_ip}"
        count = await self.redis.incr(key)
        if count == 1:
            await self.redis.expire(key, self.window)
        if count > self.max_requests:
            return JSONResponse(
                status_code=429,
                content={"code": 42901, "message": "请求过于频繁，请稍后再试"}
            )
        return await call_next(request)
```

### 3.6 认证服务 (services/auth_service.py)

```python
from datetime import datetime, timedelta
from jose import jwt, JWTError
from passlib.hash import bcrypt
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models.user import User
from app.core.config import settings

class AuthService:
    def __init__(self, db: AsyncSession):
        self.db = db

    def hash_password(self, password: str) -> str:
        return bcrypt.hash(password)

    def verify_password(self, password: str, hashed: str) -> bool:
        return bcrypt.verify(password, hashed)

    def create_access_token(self, user_id: int) -> str:
        expire = datetime.utcnow() + timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
        return jwt.encode(
            {"sub": str(user_id), "exp": expire, "type": "access"},
            settings.JWT_SECRET, algorithm=settings.JWT_ALGORITHM,
        )

    def create_refresh_token(self, user_id: int) -> str:
        expire = datetime.utcnow() + timedelta(days=settings.REFRESH_TOKEN_EXPIRE_DAYS)
        return jwt.encode(
            {"sub": str(user_id), "exp": expire, "type": "refresh"},
            settings.JWT_SECRET, algorithm=settings.JWT_ALGORITHM,
        )

    def decode_token(self, token: str) -> dict | None:
        try:
            payload = jwt.decode(token, settings.JWT_SECRET, algorithms=[settings.JWT_ALGORITHM])
            return {"user_id": int(payload["sub"]), "type": payload.get("type")}
        except JWTError:
            return None

    async def register(self, phone: str, password: str, nickname: str) -> User:
        # 检查手机号是否已注册
        existing = await self.db.execute(select(User).where(User.phone == phone))
        if existing.scalar_one_or_none():
            raise ValueError("该手机号已注册")

        user = User(
            phone=phone,
            password_hash=self.hash_password(password),
            nickname=nickname,
            role="user",
        )
        self.db.add(user)
        await self.db.commit()
        await self.db.refresh(user)
        return user

    async def login(self, phone: str, password: str) -> dict:
        result = await self.db.execute(select(User).where(User.phone == phone))
        user = result.scalar_one_or_none()
        if not user or not self.verify_password(password, user.password_hash):
            raise ValueError("手机号或密码错误")
        if user.status != "active":
            raise ValueError("账号已被禁用")

        user.last_login_at = datetime.utcnow()
        await self.db.commit()

        return {
            "user_id": user.id,
            "access_token": self.create_access_token(user.id),
            "refresh_token": self.create_refresh_token(user.id),
            "expires_in": settings.ACCESS_TOKEN_EXPIRE_MINUTES * 60,
        }

    async def wechat_login(self, code: str) -> dict:
        """微信小程序登录：code → openid → 查找/创建用户"""
        from app.utils.wechat import get_openid_by_code
        openid = await get_openid_by_code(code)

        result = await self.db.execute(select(User).where(User.wechat_openid == openid))
        user = result.scalar_one_or_none()
        is_new = False

        if not user:
            user = User(
                wechat_openid=openid,
                nickname=f"游客{openid[:6]}",
                role="user",
            )
            self.db.add(user)
            await self.db.commit()
            await self.db.refresh(user)
            is_new = True

        return {
            "user_id": user.id,
            "is_new_user": is_new,
            "access_token": self.create_access_token(user.id),
            "refresh_token": self.create_refresh_token(user.id),
        }
```

### 3.7 票务服务 (services/ticket_service.py)

```python
import uuid
from datetime import datetime, date
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models.ticket import TicketType, TicketOrder, TicketItem

class TicketService:
    def __init__(self, db: AsyncSession):
        self.db = db

    async def get_prices(self) -> list[TicketType]:
        result = await self.db.execute(
            select(TicketType).where(TicketType.status == "active").order_by(TicketType.id)
        )
        return result.scalars().all()

    async def create_order(self, user_id: int, visit_date: str, items: list[dict]) -> TicketOrder:
        # 验证游玩日期
        vd = date.fromisoformat(visit_date)
        if vd < date.today():
            raise ValueError("游玩日期不能早于今天")

        # 计算订单金额
        total = 0.0
        order_items = []
        for item in items:
            tt = await self.db.get(TicketType, item["ticket_type_id"])
            if not tt or tt.status != "active":
                raise ValueError(f"票种不存在或已下架: {item['ticket_type_id']}")
            price = tt.discount_price or tt.price
            subtotal = price * item["quantity"]
            total += subtotal
            order_items.append({
                "ticket_type_id": tt.id,
                "quantity": item["quantity"],
                "unit_price": price,
                "subtotal": subtotal,
            })

        # 创建订单
        order_no = f"T{datetime.now().strftime('%Y%m%d%H%M%S')}{uuid.uuid4().hex[:6].upper()}"
        order = TicketOrder(
            order_no=order_no,
            user_id=user_id,
            total_amount=total,
            pay_amount=total,  # 未使用优惠券时等于总额
            visit_date=vd,
            status="pending",
            pay_status="unpaid",
        )
        self.db.add(order)
        await self.db.flush()

        # 创建订单明细
        for oi in order_items:
            self.db.add(TicketItem(order_id=order.id, **oi))

        await self.db.commit()
        await self.db.refresh(order)
        return order

    async def pay_order(self, order_no: str, user_id: int) -> dict:
        """模拟支付（学习项目无需真实微信支付）"""
        result = await self.db.execute(
            select(TicketOrder).where(
                TicketOrder.order_no == order_no,
                TicketOrder.user_id == user_id,
            )
        )
        order = result.scalar_one_or_none()
        if not order:
            raise ValueError("订单不存在")
        if order.pay_status == "paid":
            raise ValueError("订单已支付")

        order.pay_status = "paid"
        order.status = "confirmed"
        order.qr_code = f"LELE-{order_no}-{uuid.uuid4().hex[:8]}"
        await self.db.commit()

        return {"order_no": order_no, "qr_code": order.qr_code, "status": "confirmed"}

    async def get_my_orders(self, user_id: int, page: int = 1, size: int = 10):
        offset = (page - 1) * size
        result = await self.db.execute(
            select(TicketOrder)
            .where(TicketOrder.user_id == user_id)
            .order_by(TicketOrder.created_at.desc())
            .offset(offset).limit(size)
        )
        return result.scalars().all()

    async def cancel_order(self, order_no: str, user_id: int) -> TicketOrder:
        result = await self.db.execute(
            select(TicketOrder).where(
                TicketOrder.order_no == order_no,
                TicketOrder.user_id == user_id,
            )
        )
        order = result.scalar_one_or_none()
        if not order:
            raise ValueError("订单不存在")
        if order.status not in ("pending", "confirmed"):
            raise ValueError(f"当前状态({order.status})不可取消")

        order.status = "cancelled"
        order.cancelled_at = datetime.utcnow()
        if order.pay_status == "paid":
            order.pay_status = "refunding"
            order.refund_amount = order.pay_amount
        await self.db.commit()
        return order
```

### 3.8 WebSocket 对话处理器 (api/ws/chat.py)

```python
import json
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Depends
from app.core.security import decode_ws_token
from app.services.dify_service import dify_service
from app.services.auth_service import AuthService

router = APIRouter()

@router.websocket("/ws/chat")
async def chat_websocket(websocket: WebSocket):
    """WebSocket 流式 AI 对话"""
    # 1. 鉴权
    token = websocket.query_params.get("token")
    if not token:
        await websocket.close(code=4001, reason="缺少 token")
        return

    payload = AuthService.decode_token(None, token)
    if not payload:
        await websocket.close(code=4001, reason="Token 无效")
        return

    user_id = str(payload["user_id"])
    await websocket.accept()

    try:
        while True:
            # 2. 接收用户消息
            raw = await websocket.receive_text()
            data = json.loads(raw)

            message = data.get("message", "").strip()
            conversation_id = data.get("conversation_id")
            app_type = data.get("app_type", "planner")  # planner/guide/ticket

            if not message:
                await websocket.send_json({"type": "error", "content": "消息不能为空"})
                continue

            # 3. 发送 thinking 状态
            await websocket.send_json({"type": "thinking", "content": "正在分析您的需求..."})

            # 4. 调用 Dify 流式 API
            full_content = ""
            conv_id = conversation_id
            try:
                async for chunk in dify_service.chat(
                    app_type=app_type,
                    query=message,
                    user_id=user_id,
                    conversation_id=conversation_id,
                    stream=True,
                ):
                    parsed = json.loads(chunk)
                    event = parsed.get("event", "")

                    if event == "message":
                        token_text = parsed.get("answer", "")
                        full_content += token_text
                        await websocket.send_json({
                            "type": "token",
                            "content": token_text,
                        })

                    if event == "message_end":
                        conv_id = parsed.get("conversation_id", conv_id)

                    if event == "error":
                        await websocket.send_json({
                            "type": "error",
                            "content": parsed.get("message", "AI 服务异常"),
                        })

            except Exception as e:
                await websocket.send_json({
                    "type": "error",
                    "content": f"Dify 调用失败: {str(e)}",
                })

            # 5. 发送完成信号
            await websocket.send_json({
                "type": "done",
                "conversation_id": conv_id,
                "full_content": full_content,
            })

    except WebSocketDisconnect:
        pass
```

## 4. 关键流程

### 4.1 AI 对话完整流程

```
用户发送消息（Web/小程序）
    │
    ▼
后端 API 接收（/api/v1/chat/message）
    │
    ▼
JWT 鉴权 → 获取用户信息
    │
    ▼
意图分类（关键词 + LLM 混合）
    │
    ├── 问候/闲聊 → 后端直接回复
    │
    ├── 推荐/路线/活动 → 调用 Dify "lele-travel-planner"
    │
    ├── 排队/设施/安全 → 调用 Dify "lele-park-guide"
    │
    └── 购票/订单 → 调用 Dify "lele-ticket-assistant"
    │
    ▼
Dify 流式返回 → WebSocket 推送给前端
    │
    ▼
保存对话记录到数据库
    │
    ▼
返回 conversation_id（续接对话用）
```

### 4.2 购票流程

```
用户选择票种 + 日期
    │
    ▼
后端创建订单（status=待支付）
    │
    ▼
调用微信支付 API 创建预付单
    │
    ▼
前端拉起微信支付
    │
    ▼
微信支付回调通知后端
    │
    ▼
后端更新订单状态（已支付）
    │
    ▼
生成电子票二维码
    │
    ▼
推送购票成功通知
    │
    ▼
入园时扫码验票
```

### 4.3 智能排队流程

```
游客到达项目入口
    │
    ▼
扫码/小程序取号
    │
    ▼
后端生成排队号，写入 Redis
    │
    ▼
前端 WebSocket 接收排队进度推送
    │
    ▼
快到号时推送提醒通知
    │
    ▼
叫号 → 游客入场
    │
    ▼
Redis 更新当前叫号
```
