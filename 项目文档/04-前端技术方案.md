## 1. 总体策略

采用 **uni-app (Vue 3)** 作为核心前端框架，一套代码同时编译为：

| 平台 | 技术 | 用途 |
|------|------|------|
| **Web 端（PC + 移动端）** | H5 编译 | 响应式网页，PC/手机浏览器访问 |
| **微信小程序** | 微信小程序编译 | 微信生态内使用，扫码即用 |
| **运营管理后台** | 独立 Vue 3 + Element Plus | 仅 PC 端，运营人员使用 |

> 核心业务代码（游客端）使用 uni-app 跨平台复用；运营管理后台因交互复杂度高，使用独立 Vue 3 项目。

## 5. 核心技术实现

### 5.1 WebSocket AI 对话

```javascript
// utils/websocket.js - uni-app WebSocket 封装
class ChatWebSocket {
  constructor(url, token) {
    this.url = url
    this.token = token
    this.ws = null
    this.handlers = {}
  }

  connect() {
    this.ws = uni.connectSocket({
      url: `${this.url}?token=${this.token}`,
      success: () => console.log('WebSocket connected'),
    })

    uni.onSocketMessage((res) => {
      const data = JSON.parse(res.data)
      this._dispatch(data)
    })

    uni.onSocketClose(() => {
      // 自动重连
      setTimeout(() => this.connect(), 3000)
    })
  }

  sendMessage(message, conversationId) {
    uni.sendSocketMessage({
      data: JSON.stringify({
        type: 'chat',
        message,
        conversation_id: conversationId,
      }),
    })
  }

  on(event, handler) {
    if (!this.handlers[event]) this.handlers[event] = []
    this.handlers[event].push(handler)
  }

  _dispatch(data) {
    const handlers = this.handlers[data.type] || []
    handlers.forEach(h => h(data))
  }
}
```

### 5.2 请求封装

```javascript
// utils/request.js - uni-app 请求封装
const BASE_URL = 'https://api.lelepark.com/api/v1'

const request = (options) => {
  return new Promise((resolve, reject) => {
    const token = uni.getStorageSync('access_token')
    uni.request({
      url: BASE_URL + options.url,
      method: options.method || 'GET',
      data: options.data,
      header: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : '',
        ...options.header,
      },
      success: (res) => {
        if (res.statusCode === 200) {
          resolve(res.data)
        } else if (res.statusCode === 401) {
          // Token 过期，刷新或重新登录
          refreshToken().then(() => request(options).then(resolve).catch(reject))
        } else {
          reject(res.data)
        }
      },
      fail: reject,
    })
  })
}

export const get = (url, data) => request({ url, method: 'GET', data })
export const post = (url, data) => request({ url, method: 'POST', data })
export const put = (url, data) => request({ url, method: 'PUT', data })
export const del = (url, data) => request({ url, method: 'DELETE', data })
```

### 5.3 Pinia 状态管理

```javascript
// stores/user.js
import { defineStore } from 'pinia'
import { post, get, put, del } from '@/utils/request'

export const useUserStore = defineStore('user', {
  state: () => ({
    token: uni.getStorageSync('access_token') || '',
    refreshToken: uni.getStorageSync('refresh_token') || '',
    userInfo: null,
    isWechat: false,
  }),
  getters: {
    isLoggedIn: (state) => !!state.token,
    isMember: (state) => state.userInfo?.member?.level > 0,
    childAge: (state) => state.userInfo?.profile?.child_age || '',
  },
  actions: {
    async login(credentials) {
      const res = await post('/auth/login', credentials)
      this.token = res.access_token
      this.refreshToken = res.refresh_token
      uni.setStorageSync('access_token', res.access_token)
      uni.setStorageSync('refresh_token', res.refresh_token)
      await this.fetchUserInfo()
    },
    async wechatLogin(code) {
      const res = await post('/auth/wechat/login', { code })
      this.token = res.access_token
      this.isWechat = true
      uni.setStorageSync('access_token', res.access_token)
      await this.fetchUserInfo()
    },
    async fetchUserInfo() {
      this.userInfo = await get('/users/me')
    },
    async updateProfile(profile) {
      await put('/users/me/profile', profile)
      await this.fetchUserInfo()
    },
    logout() {
      this.token = ''
      this.refreshToken = ''
      this.userInfo = null
      uni.removeStorageSync('access_token')
      uni.removeStorageSync('refresh_token')
    },
  },
})

// stores/chat.js — AI 对话状态管理
export const useChatStore = defineStore('chat', {
  state: () => ({
    currentConversationId: null,
    messages: [],
    isStreaming: false,
    suggestedQuestions: [],
  }),
  actions: {
    sendMessage(message) {
      this.messages.push({ role: 'user', content: message, timestamp: Date.now() })
      this.messages.push({ role: 'bot', content: '', timestamp: Date.now() })
      this.isStreaming = true
    },
    appendToken(token) {
      const lastMsg = this.messages[this.messages.length - 1]
      if (lastMsg?.role === 'bot') lastMsg.content += token
    },
    finishStream(conversationId, suggested) {
      this.currentConversationId = conversationId
      this.suggestedQuestions = suggested || []
      this.isStreaming = false
    },
    clearChat() {
      this.messages = []
      this.currentConversationId = null
      this.suggestedQuestions = []
    },
  },
})

// stores/queue.js — 排队状态管理
export const useQueueStore = defineStore('queue', {
  state: () => ({ myQueues: [], loading: false }),
  actions: {
    async fetchMyQueues() {
      this.loading = true
      this.myQueues = await get('/queue/my')
      this.loading = false
    },
    async takeNumber(projectId) {
      const res = await post('/queue/take', { project_id: projectId })
      await this.fetchMyQueues()
      return res
    },
    async cancelQueue(projectId) {
      await del(`/queue/${projectId}/cancel`)
      await this.fetchMyQueues()
    },
  },
})
```

## 6. 小程序特殊适配

### 6.1 微信登录流程

```
用户打开小程序
    │
    ▼
uni.login() 获取 code
    │
    ▼
后端 /auth/wechat/login (code)
    │
    ▼
后端调用微信 API 换取 openid + session_key
    │
    ▼
后端查找或创建用户，返回 JWT
    │
    ▼
前端存储 JWT，完成登录
```

### 6.2 微信支付流程

```
用户选择票种 → 确认订单
    │
    ▼
后端创建订单 + 微信预付单
    │
    ▼
后端返回 prepay_id
    │
    ▼
前端 uni.requestPayment({prepay_id, ...})
    │
    ▼
用户完成支付
    │
    ▼
微信支付异步通知后端
    │
    ▼
后端更新订单状态 + 生成电子票
```

## 7. 响应式设计

Web 端采用响应式布局，适配不同屏幕：

| 断点 | 宽度 | 布局策略 |
|------|------|---------|
| **Mobile** | < 768px | 单列布局，底部 TabBar 导航 |
| **Tablet** | 768-1024px | 双列布局，侧边导航 |
| **Desktop** | > 1024px | 三列布局，顶部导航 + 侧边栏 |
